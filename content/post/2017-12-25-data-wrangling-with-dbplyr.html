Show Query

mileages <- mtcars2 %>%
  group_by(cyl) %>%
  summarise(mileage = mean(mpg))

mileages %>% show_query()
## Warning: Missing values are always removed in SQL.
## Use `AVG(x, na.rm = TRUE)` to silence this warning
